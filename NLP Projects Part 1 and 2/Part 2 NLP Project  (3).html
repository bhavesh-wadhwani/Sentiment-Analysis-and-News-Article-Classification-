# Preprocessing function
def simple_preprocess(text):
    if pd.isna(text):
        return ""
    text = text.lower().translate(str.maketrans('', '', string.punctuation))
    tokens = text.split()
    tokens = [word for word in tokens if word not in stop_words]
    return ' '.join(tokens)

# Apply preprocessing
